    <script>
        // List of cooking tasks or ingredients
        const cookingItems = [
            "Make a peanut butter and jelly sandwich",
            "Bake chocolate chip cookies",
            "Peel and slice a banana",
            "Mix pancake batter",
            "Chop vegetables for a salad",
            "Boil water for pasta",
            "Make a fruit smoothie",
            "Whisk eggs for an omelette",
        ];

        // Function to generate a random cooking task or ingredient
        function generateRandomItem() {
            const randomIndex = Math.floor(Math.random() * cookingItems.length);
            return cookingItems[randomIndex];
        }

        // Event listener for the button click
        const generateButton = document.getElementById("generateButton");
        generateButton.addEventListener("click", function () {
            const output = document.getElementById("output");
            const randomItem = generateRandomItem();
            output.textContent = "Your task: " + randomItem;
        });
    </script>
